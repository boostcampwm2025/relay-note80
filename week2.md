```js
// wakeUpRoutine.test.js

const { triggerAlarm, userReactToAlarm } = require('./wakeUpRoutine');

jest.useFakeTimers();

describe('Morning Wake-up Routine', () => {
  test('User should not fall back asleep after snoozing the alarm', () => {
    // Step 1: Alarm rings
    const alarm = triggerAlarm();

    // Step 2: User snoozes
    const result = userReactToAlarm(alarm, { action: 'snooze' });

    // Simulate time passing (5 minutes of snooze)
    jest.advanceTimersByTime(5 * 60 * 1000); // 5 minutes

    // Step 3: Check user's consciousness state
    expect(result.fellBackAsleep).toBe(false);
    expect(result.actions.includes('gotUp')).toBe(true);
  });

  test('User completely ignores snooze and wakes up immediately', () => {
    const alarm = triggerAlarm();

    const result = userReactToAlarm(alarm, { action: 'wakeUpImmediately' });

    expect(result.fellBackAsleep).toBe(false);
    expect(result.actions).toContain('turnedOffAlarm');
    expect(result.actions).toContain('gotUp');
  });

  test('Fails when user falls back asleep', () => {
    const alarm = triggerAlarm();

    const result = userReactToAlarm(alarm, { action: 'snoozeThenSleep' });

    expect(result.fellBackAsleep).toBe(true);
    expect(result.actions).not.toContain('gotUp');

    // This is a failing case
    expect(() => {
      if (result.fellBackAsleep) {
        throw new Error('Regression: User fell back asleep after snoozing');
      }
    }).toThrow('Regression: User fell back asleep after snoozing');
  });
});

```

### **`[] == ![]` 의 진실 😵‍💫**

```jsx
console.log([] == ![]); // true?!
```

> 왜 true일까?
> 
> 
> `![]`는 `false` → `[] == false` → `[]`는 truthy지만 `==`는 강제 타입변환을 하니까
> 
> `[] → '' → 0`, `false → 0` → `0 == 0` → ✅ **true!**
> 

🧠 **비교 연산의 타입 강제 변환을 조심하자!**

> **🧙‍♂️`git log --oneline --graph`로 히스토리 시각화!**  
>  
> 이 명령어 하나면 Git 브랜치의 흐름이 **한눈에 그림처럼 쫙 펼쳐진다!**  
>  
> ```bash
> git log --oneline --graph --all
> ```  
>  
> 👉 커밋 메시지는 짧고 간결하게, 브랜치는 선으로 표현돼서  
> "이 브랜치가 어디서 갈라졌고, 어디서 합쳐졌는지"를 ASCII 아트처럼 볼 수 있음.  
>  
> **여러 브랜치 작업이 꼬였을 때 방향 잡는 데 딱!**  
>  
> > ✅ 꿀팁: --decorate를 추가하면 브랜치 이름도 같이 표시돼서 더 보기 편해짐!  
>  
> ```bash
> git log --oneline --graph --all --decorate
> ```

### 📌 함께 수행하고 나눈 의견

**Q1️⃣. 해당 퀘스트를 수행하는 데 AI는 어떤 역할이었나요? AI가 없으면 동일한 결과를 얻을 수 없을까요?**
- **연상훈**: 여러 꿀팁을 알려주는 것이었기 때문에, 굳이 AI가 아니였어도 큰 문제는 없었다고 생각합니다. 제가 이미 알고 있는 것 또는 구글 검색해서 나온 내용이었더라도 상관없다고 봅니다.
- **김예빈**: 개발에 활용할 수 있는 꿀팁들을 소개하는 역할. AI가 없어도 스스로 작성하거나 구글링으로 탐색할 수 있었겠지만, 원하는 주제의 개발 꿀팁들을 다양한 선택지로 제시받을 수 있어서 편리했습니다.
- **이승지**: 커뮤니티에서 공유할 내용을 제시해주는 역할. AI가 없었어도 동일한 내용은 찾을 수 있다.
- **한보원**: AI가 나에게 개발과 관련된 꿀팁을 주는 역할이었다. 진행을 하면서 생각보다 몰랐던, 숨어있는 꿀팁을 발견해내기엔 좋았다! 하지만, 정말 잘 쓰일 것 같은 꿀팁이 많지는 않아서, 선별해내는 과정이 시간이 좀 들었던것 같다. 또, 진짜 공유하고싶은 꿀팁을 찾기 위해 여러번, 다른 시간에 질문을 해보곤 했는데, 비슷한 답변이나, 사용하지 않을 듯한 내용들이 많았어서 조금 아쉬웠다.



**Q2️⃣. 해당 퀘스트는 커뮤니티 또는 학습과 성장에 도움이 되었나요?**
- **연상훈**: 커뮤니티와 학습에는 조금이나마 도움이 되지 않았을까 생각합니다. 
- **김예빈**: 유용한 정보들도 있고, 너무 가벼운 내용들도 있어서 선별이 필요했습니다. 그렇지만 개중 일부는 실제 개발에 활용해볼 수 있을만한 꿀팁이었다고 생각해요. (유용한 터미널 명령어 소개 등)
- **이승지**: AI가 제시한 꿀팁은 언제쯤 한번 써볼 수도 있을 것 같다. 내용을 공유함으로써 커뮤니티가 활성화되는데 도움이 되는 것 같다.
- **한보원**: 내가 공유한 꿀팁이 다른 사람에게 영향을 주었을진 모르겠지만,, 다른 비슷하거나 같은 퀘스트를 진행하신 동료분의 꿀팁을 보며 "오! 저런것도 있구나"하며 한번 사용해볼 수 있었다. 또, 이번 주의 두번째 과제가 git 과 관련된 미션이었는데, AI가 주는 꿀팁 중 git 명령어, 옵션이 많았어서 다양한 명령어를 접해볼 수 있었던 기회였다. 



**Q3️⃣. 해당 퀘스트는 일주일동안 수행하기에 난이도나 분량이 어땠나요?**
- **연상훈**: 난이도와 분량은 매우 적절했다고 생각합니다. 그럼에도 불구하고 부스트캠프의 미션 때문에 매일 아침 너무 지쳐 있는 상태라 매일 꾸준히 수행하진 못 했습니다.
- **김예빈**: 난이도는 평이한 편으로 수월하게 수행할 수 있었습니다. 이렇게 매일 부담없이 간단하게 AI를 활용해볼 수 있는 퀘스트가 좋은 퀘스트가 아닐까요?
- **이승지**: 난이도 하, 분량 하
- **한보원**: 수행하기에 어렵지 않았다. 미션을 하면서 퀘스트를 함께 진행하는것이 다들 쉽지 않다고 느낄 것 같은데, 그럼에도 분량과 난이도는 적당한 편이었던것 같다.



**Q4️⃣. 해당 퀘스트를 수행하며 나에게는 어떤 변화가 있었나요?**
- **연상훈**: 내가 알고 있는 생산성 기능들이 무엇이 있는지 되돌아 보고, 알고 있지만 쓰지 않았던 것들을 적극적으로 활용하고자 하는 계기가 되었습니다.
- **김예빈**: 매일 아침마다 AI가 알려주는 꿀팁을 공유하고 다른 캠퍼들의 반응을 관찰하면서 커뮤니티 활동에 참여하고 있는 느낌을 받았어요. ”캠퍼들 간의 친밀감과, 원활한 소통” 이라는 릴레이 프로젝트의 목적은 달성했다고 생각합니다!
- **이승지**: 커뮤니티 활용을 처음 해보았다.
- **한보원**: AI에게 질문하는 방법을 생각해볼 수 있는 시간이었다. AI에게 질문할때, 어떤 단어나 조사 등이 들어가느냐에 따라 답변이 많이 달라지기도, 또는 원하는 답변을 얻지 못할 수도 있다는 생각을 했다. 그래서 퀘스트를 수행하며 질문을 여러번 체크하고 나의 질문의 의도가 명확하게 전달이 되는지를 확인하고 사용하게되었다.
